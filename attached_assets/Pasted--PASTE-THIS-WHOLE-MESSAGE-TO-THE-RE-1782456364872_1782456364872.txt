// artifacts/api-server/src/routes/leads.ts
//
// Lead persistence + retrieval + CSV export for the Google Maps Lead Extractor.
// Mounted under /api/leads by routes/index.ts.
// No extra packages: uses Node built-in crypto and plain-string CSV building.

import { Router, type Request, type Response, type NextFunction } from "express";
import { createHash } from "node:crypto";
import { and, desc, gte, ilike, inArray, or, sql } from "drizzle-orm";
import { db } from "@workspace/db";
import { leads } from "@workspace/db/schema";

const router = Router();

// ---- CORS (Chrome extension calls /save cross-origin) -----------------------
router.use((_req: Request, res: Response, next: NextFunction) => {
  res.header("Access-Control-Allow-Origin", "*");
  res.header("Access-Control-Allow-Methods", "GET, POST, OPTIONS");
  res.header("Access-Control-Allow-Headers", "Content-Type");
  next();
});
router.options("*", (_req: Request, res: Response) => {
  res.sendStatus(204);
});

// ---- Types ------------------------------------------------------------------
interface IncomingLead {
  Name?: string;
  Phone?: string;
  Emails?: string;
  Website?: string;
  "Social Medias"?: string;
  Address?: string;
  Category?: string;
  Rating?: string;
  "Rating info"?: string;
  "Google Maps URL"?: string;
  "Plus Code"?: string;
  [key: string]: unknown;
}

type LeadInsert = typeof leads.$inferInsert;

// ---- Helpers ----------------------------------------------------------------
function str(value: unknown): string {
  return value == null ? "" : String(value).trim();
}

function sha256(input: string): string {
  return createHash("sha256").update(input).digest("hex");
}

// Pull the review COUNT out of "Rating info". May be a bare count ("1,234") or
// text ("4.5 stars Â· 1,234 reviews"). Never grab the rating's digits.
function parseReviewCount(ratingInfo: string): number | null {
  const cleaned = ratingInfo.replace(/,/g, "");
  const near = cleaned.match(/(\d+)\s*reviews?/i);
  if (near) return parseInt(near[1], 10);
  const ints = cleaned.replace(/\d+\.\d+/g, " ").match(/\d+/g);
  if (!ints) return null;
  return Math.max(...ints.map((n) => parseInt(n, 10)));
}

function parseRating(rating: string): number | null {
  const value = parseFloat(rating);
  return Number.isFinite(value) ? value : null;
}

// 0â€“100 lead quality score.
function computeScore(input: {
  phone: string;
  emails: string;
  website: string;
  social: string;
  rating: number | null;
  reviewCount: number | null;
  category: string;
}): number {
  let score = 0;
  if (input.phone) score += 20;
  if (input.emails) score += 20;
  if (input.website) score += 15;
  if (input.social) score += 15;
  if (input.rating != null && input.rating >= 4.0) score += 10;
  if (input.reviewCount != null && input.reviewCount >= 50) score += 10;
  if (input.category) score += 10;
  return score;
}

function toRecord(lead: IncomingLead): LeadInsert {
  const name = str(lead.Name);
  const phone = str(lead.Phone);
  const emails = str(lead.Emails);
  const website = str(lead.Website);
  const social = str(lead["Social Medias"]);
  const address = str(lead.Address);
  const category = str(lead.Category);
  const rating = parseRating(str(lead.Rating));
  const reviewCount = parseReviewCount(str(lead["Rating info"]));
  const key = sha256(`${name}${phone}${address}`);
  const score = computeScore({ phone, emails, website, social, rating, reviewCount, category });

  return {
    key,
    name: name || null,
    phone: phone || null,
    emails: emails || null,
    website: website || null,
    social: social || null,
    address: address || null,
    category: category || null,
    rating: rating != null ? String(rating) : null, // numeric() maps to string in Drizzle
    reviewCount,
    score,
    gmapsUrl: str(lead["Google Maps URL"]) || null,
    plusCode: str(lead["Plus Code"]) || null,
    raw: lead as Record<string, unknown>,
  };
}

function chunk<T>(items: T[], size: number): T[][] {
  const out: T[][] = [];
  for (let i = 0; i < items.length; i += size) out.push(items.slice(i, i + size));
  return out;
}

// ---- POST /save -------------------------------------------------------------
router.post("/save", async (req: Request, res: Response) => {
  const body = req.body;
  if (!Array.isArray(body)) {
    return res.status(400).json({ error: "Body must be an array of leads." });
  }

  const incoming = body.slice(0, 1000) as IncomingLead[];

  const byKey = new Map<string, LeadInsert>();
  for (const lead of incoming) {
    const record = toRecord(lead);
    if (record.key) byKey.set(record.key, record);
  }
  const records = [...byKey.values()];

  if (records.length === 0) {
    return res.json({ saved: 0, duplicates: incoming.length });
  }

  const keys = records.map((r) => r.key);
  const existingRows = await db
    .select({ key: leads.key })
    .from(leads)
    .where(inArray(leads.key, keys));
  const existing = new Set(existingRows.map((r) => r.key));

  for (const part of chunk(records, 500)) {
    await db
      .insert(leads)
      .values(part)
      .onConflictDoUpdate({
        target: leads.key,
        set: {
          name: sql`excluded.name`,
          phone: sql`excluded.phone`,
          emails: sql`excluded.emails`,
          website: sql`excluded.website`,
          social: sql`excluded.social`,
          address: sql`excluded.address`,
          category: sql`excluded.category`,
          rating: sql`excluded.rating`,
          reviewCount: sql`excluded.review_count`,
          score: sql`excluded.score`,
          gmapsUrl: sql`excluded.gmaps_url`,
          plusCode: sql`excluded.plus_code`,
          raw: sql`excluded.raw`,
          updatedAt: sql`now()`,
        },
      });
  }

  const saved = records.filter((r) => !existing.has(r.key)).length;
  const duplicates = incoming.length - saved;

  req.log.info({ saved, duplicates, received: incoming.length }, "leads saved");
  return res.json({ saved, duplicates });
});

// ---- GET / (paginated list) -------------------------------------------------
router.get("/", async (req: Request, res: Response) => {
  const page = Math.max(1, parseInt(str(req.query.page), 10) || 1);
  const limit = Math.min(200, Math.max(1, parseInt(str(req.query.limit), 10) || 50));
  const search = str(req.query.search);
  const minScore = parseInt(str(req.query.minScore), 10) || 0;
  const offset = (page - 1) * limit;

  const conditions = [gte(leads.score, minScore)];
  if (search) {
    const term = `%${search}%`;
    const match = or(ilike(leads.name, term), ilike(leads.address, term));
    if (match) conditions.push(match);
  }
  const where = and(...conditions);

  const [{ count }] = await db
    .select({ count: sql<number>`count(*)::int` })
    .from(leads)
    .where(where);

  const rows = await db
    .select()
    .from(leads)
    .where(where)
    .orderBy(desc(leads.score), desc(leads.createdAt))
    .limit(limit)
    .offset(offset);

  const total = count ?? 0;
  req.log.info({ page, limit, total, search, minScore }, "leads listed");
  return res.json({
    leads: rows,
    total,
    page,
    pages: Math.max(1, Math.ceil(total / limit)),
  });
});

// ---- GET /export.csv --------------------------------------------------------
function csvCell(value: unknown): string {
  const s = value == null ? "" : String(value);
  return /[",\n\r]/.test(s) ? `"${s.replace(/"/g, '""')}"` : s;
}

router.get("/export.csv", async (req: Request, res: Response) => {
  const search = str(req.query.search);
  const minScore = parseInt(str(req.query.minScore), 10) || 0;

  const conditions = [gte(leads.score, minScore)];
  if (search) {
    const term = `%${search}%`;
    const match = or(ilike(leads.name, term), ilike(leads.address, term));
    if (match) conditions.push(match);
  }

  const rows = await db
    .select()
    .from(leads)
    .where(and(...conditions))
    .orderBy(desc(leads.score), desc(leads.createdAt));

  const headers = [
    "Name", "Phone", "Emails", "Website", "Social", "Address", "Category",
    "Rating", "Reviews", "Score", "Google Maps URL", "Plus Code",
  ];

  const lines: string[] = [headers.join(",")];
  for (const r of rows) {
    lines.push([
      r.name, r.phone, r.emails, r.website, r.social, r.address, r.category,
      r.rating, r.reviewCount, r.score, r.gmapsUrl, r.plusCode,
    ].map(csvCell).join(","));
  }
  const csv = lines.join("\r\n");

  const date = new Date().toISOString().slice(0, 10);
  res.setHeader("Content-Type", "text/csv; charset=utf-8");
  res.setHeader("Content-Disposition", `attachment; filename="leads-${date}.csv"`);
  req.log.info({ rows: rows.length, search, minScore }, "leads exported to csv");
  return res.send(csv);
});

export default router;
